def test_body
  output = table_for([@drummer3, @drummer4]) do |t|
    t.body do |e|
      t.r do
        concat t.d(e.id)
        concat t.d(e.name)
      end
    end
  end
  expected = %Q(<table>) <<
      %Q(<tbody>) <<
        %Q(<tr>) <<
          %Q(<td>3</td>) <<
          %Q(<td>Peter &quot;James&quot; Bond</td>) <<
        %Q(</tr>) <<
        %Q(<tr>) <<
          %Q(<td>4</td>) <<
          %Q(<td>Mick Shrimpton (R. J. &quot;Ric&quot; Parnell)</td>) <<
        %Q(</tr>) <<
      %Q(</tbody>) <<
    %Q(</table>)
  assert_dom_equal expected, output
end
